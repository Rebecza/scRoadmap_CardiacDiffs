source_oi <- c("ZNF711","ZNF503","LHX2")
for (soi in source_oi){
  for (network_subset in network_subsets_files){
    if (grepl(soi, network_subset)){
      print(TRUE)
      subset_df <- read.table(network_subset, header = T)
      tmp <- as.data.frame(str_split_fixed(subset_df$tf_target, "", 2))
      colnames(tmp) <- c("source","target")
      
      subset_df <- subset_df[!colnames(subset_df) %in% "tf_target"]
    
      tmp <- cbind(tmp, subset_df)
      tmp <- tmp[tmp$weighted_binding>0,]
      tmp <- tmp[!grepl("-AS", tmp$target) | grepl("-AS", tmp$source),]
      tmp <- tmp[!grepl("-DT", tmp$target) | grepl("-DT", tmp$source),]
      
      toi_df <- tmp
      print(paste0("The amount of interactions found with ", soi, " is ", length(rownames(toi_df))))
      hist(toi_df$weighted_binding, breaks=100, main=paste0("weighted_binding ", soi))
      hist(toi_df$prob, breaks=100, main=paste0("prob ", soi))
      write.table(toi_df, paste0(resultsdir,"/NetworkOfSource_",soi,".tsv"), quote = F, row.names = F)
    }
  }
}
