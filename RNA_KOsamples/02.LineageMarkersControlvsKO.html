## Save list with rld_values, to plot hits in another RMD.
rld <- readRDS(paste0(rld_path))
seuset.markers <- read.csv(atlas_marker_path, row.names = 1)
top_markers <- seuset.markers %>% group_by(cluster) %>% top_n(n = no_of_DEgenes, wt = avg_log2FC)

top_markers <- top_markers[,c("gene","cluster"), drop = F]
marker_list <- list()
for (i in unique(top_markers$cluster)){
  marker_list[[i]] <- top_markers[top_markers$cluster == i,]$gene
}

Create list per cell type

#mat <- t(scale(t(assay(rld))))
mat <- assay(rld)

marker_expr_list <- list()

for (i in names(marker_list)){
  marker_expr_list[[i]] <- mat[rownames(mat) %in% marker_list[[i]],]
}

Paired Sign-test; Wilcox paired = T

# mat_to_plot
sample_names <- unique(gsub("_[0-9]","",colnames(mat)))
wilcox_table <- as.data.frame(matrix(nrow = length(names(marker_list)),ncol = length(colnames(mat))))
rownames(wilcox_table) <- names(marker_list)
colnames(wilcox_table) <- colnames(mat)
padj_table <- wilcox_table
ko_samples <- colnames(mat)[!grepl("TRAC", colnames(mat))]

for (i in names(marker_expr_list)){
  for (j in ko_samples){
    ct_lineage <- unique(gsub("_.*_[0-9]","",j))
    ct_sample <- unique(gsub(".*_.*_","",j))
    ct_name <- paste0(ct_lineage, "_TRAC_", ct_sample)
    print(paste0("Wilcox sign test will be performed paired on: ",ct_name, " and ", j))
    identical(names(marker_expr_list[[i]][,ct_name]),names(marker_expr_list[[i]][,j]))
    wilcox <- wilcox.test(x = marker_expr_list[[i]][,ct_name], marker_expr_list[[i]][,j], paired = T)
    
    wilcox_table[i,j] <- wilcox$p.value 
    padj_table[i,j] <- p.adjust(wilcox$p.value, method = "BH", n=length(sample_names[!grepl("TRAC",sample_names)]))
  }
}
